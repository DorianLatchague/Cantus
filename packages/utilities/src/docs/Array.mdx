# Array Utilities

Immutable helper functions for working with arrays. Each function returns a new array
rather than mutating the original, making them ideal for use in React state updates.

All functions are exported from `@cantus/utilities`.

```ts
import { replaceInArray, removeFirst, upsertToEnd, upsertToStart } from "@cantus/utilities";
```

---

## `replaceInArray`

Returns a shallow copy of an array with the first item matching a predicate replaced.

### Signature

```ts
replaceInArray<T>(
  items: T[],
  update: T | ((item: T) => T),
  predicate: (item: T) => boolean,
): T[];
```

### Parameters

| Parameter   | Type                          | Description                                        |
| ----------- | ----------------------------- | -------------------------------------------------- |
| `items`     | `T[]`                         | The source array.                                  |
| `update`    | `T \| ((item: T) => T)`      | The replacement value, or a function that receives the matched item and returns the replacement. |
| `predicate` | `(item: T) => boolean`        | A function to find the item to replace.            |

### Example

```ts
const users = [{ id: 1, name: "Alice" }, { id: 2, name: "Bob" }];

// Replace with a static value
replaceInArray(users, { id: 2, name: "Robert" }, (u) => u.id === 2);
// => [{ id: 1, name: "Alice" }, { id: 2, name: "Robert" }]

// Replace with an updater function
replaceInArray(users, (u) => ({ ...u, name: "Robert" }), (u) => u.id === 2);
// => [{ id: 1, name: "Alice" }, { id: 2, name: "Robert" }]
```

> If no item matches the predicate, the **original** array reference is returned.

## `removeFirst`

Returns a shallow copy of an array with the first item matching a predicate removed.

### Signature

```ts
removeFirst<T>(
  items: T[],
  predicate: (item: T) => boolean,
  onRemove?: (removedItem: T) => void,
): T[];
```

### Parameters

| Parameter   | Type                          | Description                                        |
| ----------- | ----------------------------- | -------------------------------------------------- |
| `items`     | `T[]`                         | The source array.                                  |
| `predicate` | `(item: T) => boolean`        | A function to find the item to remove.             |
| `onRemove`  | `(removedItem: T) => void`    | Optional callback invoked with the removed item.   |

### Example

```ts
const items = ["apple", "banana", "cherry"];

removeFirst(items, (i) => i === "banana");
// => ["apple", "cherry"]

removeFirst(items, (i) => i === "banana", (removed) => {
  console.log(`Removed: ${removed}`); // "Removed: banana"
});
```

> If no item matches, the **original** array reference is returned.

## `upsertToEnd`

Replaces the first item matching the predicate, or appends the item to the **end** of the array.

### Signature

```ts
upsertToEnd<T>(
  items: T[],
  item: T,
  predicate: (item: T) => boolean,
): T[];
```

### Example

```ts
const tags = [{ id: 1, label: "ts" }];

// Update existing
upsertToEnd(tags, { id: 1, label: "typescript" }, (t) => t.id === 1);
// => [{ id: 1, label: "typescript" }]

// Insert new (appended)
upsertToEnd(tags, { id: 2, label: "react" }, (t) => t.id === 2);
// => [{ id: 1, label: "ts" }, { id: 2, label: "react" }]
```

---

## `upsertToStart`

Replaces the first item matching the predicate, or prepends the item to the **start** of the array.

### Signature

```ts
upsertToStart<T>(
  items: T[],
  item: T,
  predicate: (item: T) => boolean,
): T[];
```

### Example

```ts
const tags = [{ id: 1, label: "ts" }];

// Update existing
upsertToStart(tags, { id: 1, label: "typescript" }, (t) => t.id === 1);
// => [{ id: 1, label: "typescript" }]

// Insert new (prepended)
upsertToStart(tags, { id: 2, label: "react" }, (t) => t.id === 2);
// => [{ id: 2, label: "react" }, { id: 1, label: "ts" }]
```
